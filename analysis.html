grouped_data <- clean_data %>%
  group_by(Race = Race...Ethnicity, Income = Income.Groups) %>%

total_weights_per_race <- grouped_data %>%
  group_by(Race) %>%
  summarise(Total_Weight = sum(Weighted_Count), .groups = 'drop')
